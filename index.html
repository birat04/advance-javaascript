async function fetchPosts() { try { const response = await axios.get("https://jsonplaceholder.typicode.com/posts"); const posts = response.data; const postsContainer = document.getElementById("posts"); postsContainer.innerHTML = ""; posts.forEach(post => { const postElement = document.createElement("div"); postElement.textContent = post.body; postElement.className = "post"; // Optional: Add a class for styling postsContainer.appendChild(postElement); }); } catch (error) { console.error("Error fetching posts:", error); } }
